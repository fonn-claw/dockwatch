    **Step 1: Create src/lib/auth/session.ts**
    ```typescript
    import { SessionOptions } from "iron-session";

    export interface SessionData {
      userId: number;
      email: string;
      name: string;
      role: "manager" | "crew" | "inspector";
      isLoggedIn: boolean;
    }

    export const sessionOptions: SessionOptions = {
      password: process.env.SESSION_SECRET!,
      cookieName: "dockwatch-session",
      ttl: 60 * 60 * 24 * 7, // 7 days
      cookieOptions: {
        secure: process.env.NODE_ENV === "production",
        httpOnly: true,
        sameSite: "lax" as const,
      },
    };

    export async function getSession() {
      const { cookies } = await import("next/headers");
      const { getIronSession } = await import("iron-session");
      return getIronSession<SessionData>(await cookies(), sessionOptions);
    }
    ```

    **Step 2: Create src/lib/auth/guards.ts**
    ```typescript
    import { getSession, SessionData } from "./session";
    import { redirect } from "next/navigation";

    export async function requireAuth(): Promise<SessionData> {
      const session = await getSession();
      if (!session.isLoggedIn) {
        redirect("/login");
      }
      return session as SessionData;
    }

    export async function requireRole(
      allowedRoles: SessionData["role"][]
    ): Promise<SessionData> {
      const session = await requireAuth();
      if (!allowedRoles.includes(session.role)) {
        redirect("/dashboard");
      }
      return session;
    }
    ```
    Note: unauthorized redirect goes to /dashboard (not /unauthorized) -- simpler for demo, avoids needing a dedicated 403 page.

    **Step 3: Create src/lib/actions/auth.ts**
    Server actions with "use server" directive:
    - `login(formData: FormData)`: Validates with Zod (email: z.string().email(), password: z.string().min(1)), queries users table by email via Drizzle, compares password with bcryptjs `compare()`, sets session fields (userId, email, name, role, isLoggedIn=true), calls `await session.save()`, then redirects based on role: manager->/dashboard, crew->/work-orders, inspector->/compliance. Returns `{ error: "Invalid email or password" }` on failure (generic message, no enumeration).
    - `logout()`: Gets session via getSession(), calls session.destroy(), redirects to /login.
