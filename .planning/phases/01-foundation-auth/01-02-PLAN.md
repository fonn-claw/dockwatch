      { label: "Audit Trail", href: "/compliance/audit", icon: FileText, roles: ["inspector"] },
      { label: "Cost Reports", href: "/reports", icon: DollarSign, roles: ["manager"] },

    Each placeholder page layout:
    ```tsx
    import { requireRole } from "@/lib/auth/guards";
    import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
    import { IconName } from "lucide-react"; // matching icon from nav-config

    export default async function PageName() {
      await requireRole([...allowedRoles]);

      return (
        <div className="space-y-6">
          <div>
            <h1 className="text-3xl font-bold tracking-tight">Page Title</h1>
            <p className="text-muted-foreground">Brief description</p>

            </CardContent>
          </Card>
        </div>
      );
    }
    ```
